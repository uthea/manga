<script type="text/javascript">
function prepareWidget(args) {
  var Widget = args || {}
  Widget.template = document.querySelector(Widget.template_selector).innerHTML
  Widget.impressions = []
  var removeWidget = function() {
    jQuery(Widget.hidden_id).remove()
  }
  var toBoolean = function(str) {
    return str === 'true'
  }
  var addWidgetImpressions = function(item) {
    Widget.impressions.push({
      id: item.master_code,
      list: Widget.ga_list_name,
      position: item.position,
      coupon: item.data_coupon
    })
  }
  var flagTransformWithCurrentDate = function(item) {
    var isInDuration = function(itemStartAt, itemEndAt) {
      var startDate = new Date(itemStartAt)
      var startFlag = Widget.current_date >= startDate
      if (itemEndAt) {
        var endDate = new Date(itemEndAt)
        var endFlag = Widget.current_date < endDate
        return  startFlag && endFlag
      } else {
        return startFlag
      }
    }
    if (item.discount_flag) {
      if (!isInDuration(item.discount_start_at, item.discount_end_at)) item.discount_flag = false
    }
    if (item.always_discount_flag) {
      if (!isInDuration(item.always_discount_start_at)) item.always_discount_flag = false
    }
    if (item.free_daily_flag) {
      if (!isInDuration(item.free_daily_start_at, item.free_daily_end_at)) item.free_daily_flag = false
    }
  }
  var setButtonAttributes = function(item) {
    var isButtonShow = function(item) {
      return (item.sample_flag || item.discount_flag || item.always_discount_flag || item.free_daily_flag)
    }
    item.data_pdc_action = 'not_set'
    item.data_coupon = ''
    if (item.sample_flag) {
      item.data_pdc_action = 'sample'
      item.button_name = '無料試し読み'
      item.button_href = item.url
      item.button_div_class = 'btn_sample'
      item.data_coupon = '試し読み'
    }
    if (item.discount_flag) {
      item.data_pdc_action = 'discount'
      item.button_name = item.discount_max_number + '話無料'
      item.button_href = item.url
      item.button_div_class = 'btn_free'
      item.data_coupon = '期間限定無料'
    }
    if (item.always_discount_flag) {
      item.data_pdc_action = 'always_discount'
      item.button_name = item.always_discount_max_number + '話無料'
      item.button_href = item.url
      item.button_div_class = 'btn_free'
      item.data_coupon = '常時無料'
    }
    if (item.free_daily_flag) {
      item.data_pdc_action = 'free_daily'
      item.button_name = item.free_daily_max_chapter_number_for_free + '話無料'
      item.button_href = item.url
      item.button_div_class = 'btn_free'
      if (item.free_daily_standard_flag) {
        item.data_coupon = '毎日無料_通常版'
      } else if (item.free_daily_short_flag) {
        item.data_coupon = '毎日無料_時短版'
      }
    }
    item.button_flag = isButtonShow(item)
  }
  var itemsTransform = function(items) {
    var dataParent = function(item) {
      return 'ec_' + Widget.ga_list_name + '_' + item.position
    }
    var dataPdcEvid = function(item) {
      return 'ev_' + Widget.ga_list_name + '_' + item.position + '_product_click'
    }
    for(var i = 0; i < items.length; i++) {
      items[i].position = i + 1 //data-position attribute
      items[i].data_list = Widget.ga_list_name
      items[i].data_id = items[i].master_code
      items[i].data_name = items[i].title
      items[i].data_parent = dataParent(items[i])
      items[i].novel_flag = toBoolean(items[i].novel_flag)
      items[i].webtoon_flag = toBoolean(items[i].webtoon_flag)
      items[i].sample_flag = toBoolean(items[i].sample_flag)
      items[i].discount_flag = toBoolean(items[i].discount_flag)
      items[i].always_discount_flag = toBoolean(items[i].always_discount_flag)
      items[i].new_book_flag = toBoolean(items[i].new_book_flag)
      items[i].new_chapter_flag = toBoolean(items[i].new_chapter_flag)
      items[i].free_daily_flag = toBoolean(items[i].free_daily_flag)
      items[i].free_daily_standard_flag = toBoolean(items[i].free_daily_standard_flag)
      items[i].free_daily_short_flag = toBoolean(items[i].free_daily_short_flag)
      items[i].data_pdc_evid = dataPdcEvid(items[i])
      items[i].review_score_flag = Widget.review_score_flag
      if (Widget.review_score_flag) {
        items[i].reviews_average_score = items[i].reviews_average_score.toPrecision(2);
      }
      flagTransformWithCurrentDate(items[i])
      setButtonAttributes(items[i])
      addWidgetImpressions(items[i])
    }
    return items
  }
  var itemsGrouping = function(items, size) {
    if (size <= 0) size = items.length;

    var groups = [];
    var index = 0;
    var groupIndex = 0;

    while (index < items.length) {
      // tag_groupの時、data-listにgroup毎のindex+1を付与する
      groupIndex++;
      var targetItems = items.slice(index, index + size);
      targetItems.forEach(function(item) {
        item.data_list = item.data_list + "_" + groupIndex;
      });
      groups.push({title_item: items[index], group_items: targetItems});
      index += size;
    }
    return groups;
  }
  var trackGa = function(items) {
    var trackGaSelector = Widget.tracking_selector + ' .js-li_track'
    ga_ecommerce_view_item_list(JSON.stringify(Widget.impressions))
    if (Widget.ga_ecommerce_target_user_flag) {
      ga_ecommerce_product_click(trackGaSelector)
    } else {
      ga_product_click_event(trackGaSelector)
    }
  }
  var registerCallback = function(resp) {
    try {
      var el = document.querySelector(Widget.selector)
      resp.items = itemsTransform(resp.items)
      if (Widget.grouping_size) {
        resp.groups = itemsGrouping(resp.items, Widget.grouping_size)
      }
      el.innerHTML = $p('render', Widget.template, resp)
      trackGa(resp.items)
      if (typeof observerController !== "undefined") {
        const idSelector = Widget.tracking_selector + " .js-li_track";
      }

      $p('track', {
        elements: document.querySelectorAll(Widget.tracking_selector),
        name: Widget.name,
        source: 'LI'
      })
    } catch(e) {
      removeWidget()
    }
  }
  var execRecommend = function() {
    $p('register', {
      max: Widget.item_max_size,
      widget: Widget.name,
      opts: Widget.register_opts, // Replace with more specific opts to invoke filter rules, once they are set up
      callback: registerCallback
    })
    $p('setRequestFields', ['url','title','jacket_image_class','jacket_image_alt','jacket_image_path','master_code',
       'book_id','first_chapter_id','novel_flag','sample_flag','webtoon_flag','always_discount_flag',
       'discount_flag','first_chapter_id','discount_max_number','always_discount_max_number','free_chapter_id',
       'discount_start_at','discount_end_at','always_discount_start_at',
       'free_daily_flag','free_daily_standard_flag','free_daily_short_flag','free_daily_start_at','free_daily_end_at','free_daily_max_chapter_number_for_free', 'new_book_flag', 'new_chapter_flag', 'tag_group_label', 'tag_group_name',
       'reviews_average_score'])
    if (Widget.multi_fetch) {
      $p('fetch')
    }
  }
  var abTestSliceCallback = function(abSlice) {
    try {
      if (abSlice < Widget.lift_igniter_slice) {
        execRecommend()
      } else {
        removeWidget()
      }
    } catch(e) {
      removeWidget()
    }
  }
  if (Widget.ab_test_flag) {
    $p('abTestSlice', {
